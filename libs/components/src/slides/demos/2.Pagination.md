# en-US

Use the `pagination` prop to set the pagination.

# zh-CN

使用 `pagination` 属性设置分页器。

```tsx
import { Compose, Radio, Separator, Slides, Switch } from '@laser-ui/components';
import { useImmer } from '@laser-ui/hooks';

export default function Demo() {
  const [pagination, setPagination] = useImmer({
    visible: true,
    dynamic: false,
  });

  return (
    <>
      <h5>visible</h5>
      <Radio.Group
        list={['true', 'false', 'hover'].map((pagination) => ({
          label: pagination,
          value: pagination,
        }))}
        model={pagination.visible.toString()}
        pattern="outline"
        onModelChange={(value) => {
          setPagination((draft) => {
            draft.visible = value === 'true' ? true : value === 'false' ? false : value;
          });
        }}
      >
        {(groupProps, optionProps, options) => (
          <Compose {...groupProps} className="mb-3">
            {options.map((opt) => (
              <Radio key={opt.value} {...optionProps(opt)} />
            ))}
          </Compose>
        )}
      </Radio.Group>
      <br />
      <h5>dynamic</h5>
      <Switch
        model={pagination.dynamic}
        onModelChange={(dynamic) => {
          setPagination((draft) => {
            draft.dynamic = dynamic;
          });
        }}
      />
      <Separator />
      <Slides
        list={Array.from({ length: 8 }).map((_, index) => ({
          id: index,
          content: (
            <div className="slide" style={{ background: index % 2 ? 'rgb(54 77 121)' : 'rgb(94 113 148)' }}>
              Slide {index}
            </div>
          ),
        }))}
        pagination={pagination}
      />
    </>
  );
}
```

```scss
h5 {
  margin: 0 0 8px;
}

.slide {
  display: flex;
  align-items: center;
  justify-content: center;
  width: 100%;
  height: 160px;
  color: #fff;
}
```
