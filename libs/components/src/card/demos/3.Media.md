# en-US

This shows how to embed an image.

# zh-CN

这里展示了如何嵌入图像。

```tsx
import { Card, Icon, Separator, Switch } from '@laser-ui/components';
import { ReactComponent as DeleteOutlined } from '@material-design-icons/svg/outlined/delete.svg';
import { ReactComponent as EditOutlined } from '@material-design-icons/svg/outlined/edit.svg';
import { ReactComponent as MoreHorizOutlined } from '@material-design-icons/svg/outlined/more_horiz.svg';
import { useState } from 'react';

export default function Demo() {
  const [actions, setActions] = useState(false);

  return (
    <>
      <Switch model={actions} onModelChange={setActions}>
        actions
      </Switch>
      <Separator />
      <Card
        style={{ width: 300 }}
        actions={
          actions
            ? [
                <Card.Action title="edit" disabled>
                  <Icon>
                    <EditOutlined />
                  </Icon>
                </Card.Action>,
                <Card.Action title="delete">
                  <Icon>
                    <DeleteOutlined />
                  </Icon>
                </Card.Action>,
                <Card.Action title="more">
                  <Icon>
                    <MoreHorizOutlined />
                  </Icon>
                </Card.Action>,
              ]
            : undefined
        }
      >
        <div style={{ margin: '-1px -1px 0px -1px' }}>
          <img
            style={{ display: 'block', width: '100%', borderRadius: 'var(--l-border-radius) var(--l-border-radius) 0 0' }}
            src="/imgs/bg-1.png"
            alt="bg-1"
          />
        </div>
        <Card.Content>
          <div>Some contents...</div>
          <div>Some contents...</div>
          <div>Some contents...</div>
        </Card.Content>
      </Card>
    </>
  );
}
```
