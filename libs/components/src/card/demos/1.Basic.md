# zh-CN

最简单的用法。

```tsx
import { Button, Card, Icon, Separator, Switch } from '@laser-ui/components';
import { ReactComponent as DeleteOutlined } from '@material-design-icons/svg/outlined/delete.svg';
import { ReactComponent as EditOutlined } from '@material-design-icons/svg/outlined/edit.svg';
import { ReactComponent as MoreHorizOutlined } from '@material-design-icons/svg/outlined/more_horiz.svg';
import { useState } from 'react';

export default function Demo() {
  const [actions, setActions] = useState(false);

  return (
    <>
      <Switch model={actions} onModelChange={setActions}>
        actions
      </Switch>
      <Separator />
      <Card
        style={{ width: 300 }}
        header={<Card.Header action={<Button pattern="link">More</Button>}>Card title</Card.Header>}
        actions={
          actions
            ? [
                <Card.Action title="edit" disabled>
                  <Icon>
                    <EditOutlined />
                  </Icon>
                </Card.Action>,
                <Card.Action title="delete">
                  <Icon>
                    <DeleteOutlined />
                  </Icon>
                </Card.Action>,
                <Card.Action title="more">
                  <Icon>
                    <MoreHorizOutlined />
                  </Icon>
                </Card.Action>,
              ]
            : undefined
        }
      >
        <Card.Content>
          <div>Some contents...</div>
          <div>Some contents...</div>
          <div>Some contents...</div>
        </Card.Content>
      </Card>
    </>
  );
}
```
