# en-US

The basic usage of `Upload.PictureList`.

# zh-CN

`Upload.PictureList` 的基本用法。

```tsx
import { Button, Icon, Upload } from '@laser-ui/components';
import { ReactComponent as FileUploadOutlined } from '@material-design-icons/svg/outlined/file_upload.svg';

export default function Demo() {
  return (
    <Upload
      defaultModel={[
        {
          uid: 1,
          name: 'logo.svg',
          state: 'load',
          url: '/logo.png',
          thumbUrl: '/logo.png',
        },
        {
          uid: 2,
          name: 'logo.svg',
          state: 'load',
          url: '/logo.png',
        },
        {
          uid: 3,
          name: 'react-devui.png',
          state: 'error',
        },
        {
          uid: 4,
          name: 'react-devui.png',
          state: 'progress',
          percent: 30,
        },
      ]}
      request={{ url: 'https://laser-ui.free.beeceptor.com' }}
    >
      {(render) => (
        <>
          {render(
            <Button
              className="mb-2"
              icon={
                <Icon>
                  <FileUploadOutlined />
                </Icon>
              }
            >
              Upload
            </Button>,
          )}
          <Upload.PictureList />
        </>
      )}
    </Upload>
  );
}
```
