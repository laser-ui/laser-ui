# en-US

The basic usage of `Upload.Picture`.

# zh-CN

`Upload.Picture` 的基本用法。

```tsx
import { Upload } from '@laser-ui/components';
import { useState } from 'react';

export default function Demo() {
  const [files, setFiles] = useState([
    {
      uid: 1,
      name: 'logo.svg',
      state: 'load',
      url: '/logo.png',
      thumbUrl: '/logo.png',
    },
    {
      uid: 2,
      name: 'logo.svg',
      state: 'load',
      url: '/logo.png',
    },
    {
      uid: 3,
      name: 'react-devui.png',
      state: 'error',
    },
    {
      uid: 4,
      name: 'react-devui.png',
      state: 'progress',
      percent: 30,
    },
  ]);

  return (
    <Upload
      model={files}
      request={{ url: 'https://laser-ui.free.beeceptor.com' }}
      onModelChange={(files) => {
        setFiles(files);
      }}
    >
      {(render) => <Upload.Picture>{files.length >= 6 ? null : render(<Upload.Button />)}</Upload.Picture>}
    </Upload>
  );
}
```
