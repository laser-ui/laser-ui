# en-US

The basic usage of `Upload.Button`.

# zh-CN

`Upload.Button` 的基本用法。

```tsx
import { DialogService, Upload } from '@laser-ui/components';
import { useState } from 'react';

export default function Demo() {
  const [file, setFile] = useState();

  return (
    <Upload
      request={{ url: 'https://laser-ui.free.beeceptor.com' }}
      beforeUpload={(file) => {
        const checkType = file.type === 'image/jpeg' || file.type === 'image/png';
        if (!checkType) {
          const { close } = DialogService.open(Toast, {
            children: `Only support JPG/PNG file!`,
            onClose: () => {
              close();
            },
          });
        }

        const checkSize = file.size / 1024 / 1024 < 2;
        if (!checkSize) {
          const { close } = DialogService.open(Toast, {
            children: `Only support size that smaller than 2MB!`,
            onClose: () => {
              close();
            },
          });
        }

        return checkType && checkSize;
      }}
      onModelChange={(files, data) => {
        const file = data.files[0];
        if (file.state !== null) {
          setFile(file);
        }
      }}
    >
      <Upload.Button
        file={file}
        onRemove={() => {
          setFile(undefined);
        }}
      />
    </Upload>
  );
}
```
