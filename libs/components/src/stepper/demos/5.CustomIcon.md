# en-US

This shows how to customize dots of the component.

# zh-CN

这里展示了如何自定义组件轴点。

```tsx
import { Icon, Separator, Stepper } from '@laser-ui/components';
import { ReactComponent as PhoneEnabledOutlined } from '@material-design-icons/svg/outlined/phone_enabled.svg';
import { ReactComponent as SmsOutlined } from '@material-design-icons/svg/outlined/sms.svg';

export default function Demo() {
  return (
    <>
      <Stepper
        active={2}
        list={[
          {
            title: 'Step 1',
            description: 'This is 1 description.',
            dot: false,
          },
          {
            title: 'Step 2',
            description: 'This is 2 description.',
            color: 'rgb(123 74 203)',
            dot: false,
          },
          {
            title: 'Step 3',
            description: 'This is 3 description.',
            dot: false,
          },
        ]}
        dotSize={8}
      />
      <Separator />
      <Stepper
        active={2}
        list={[
          {
            title: 'Step 1',
            description: 'This is 1 description.',
            dot: (
              <Icon>
                <PhoneEnabledOutlined />
              </Icon>
            ),
          },
          {
            title: 'Step 2',
            description: 'This is 2 description.',
            dot: (
              <Icon>
                <SmsOutlined />
              </Icon>
            ),
          },
          {
            title: 'Step 3',
            description: 'This is 3 description.',
          },
        ]}
      />
    </>
  );
}
```
