# en-US

How to customize step states.

# zh-CN

如何自定义步骤状态。

```tsx
import { Stepper } from '@laser-ui/components';

export default function Demo() {
  return (
    <Stepper
      active={3}
      list={[
        {
          title: 'Step 1',
          description: 'This is 1 description.',
        },
        {
          title: 'Step 2',
          state: 'error',
          description: 'This is 2 description.',
        },
        {
          title: 'Step 3',
          description: 'This is 3 description.',
        },
      ]}
    />
  );
}
```
