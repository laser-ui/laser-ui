# en-US

The basic usage of `Image.Preview`.

# zh-CN

`Image.Preview` 的基本用法。

```tsx
import { Icon, Image } from '@laser-ui/components';
import { ReactComponent as VisibilityOutlined } from '@material-design-icons/svg/outlined/visibility.svg';
import { useState } from 'react';

const imgs = Array.from({ length: 4 }).map((_, i) => `/imgs/image-${i + 1}.jpg`);
export default function Demo() {
  const [active, setActive] = useState(0);
  const [visible, setVisible] = useState(false);

  return (
    <>
      <div className="row g-2">
        {imgs.map((src, index) => (
          <div key={src} className="col-auto">
            <Image
              style={{ height: 100 }}
              imgProps={{ src }}
              actions={[
                <Image.Action
                  className="button"
                  onClick={() => {
                    setActive(index);
                    setVisible(true);
                  }}
                >
                  <Icon>
                    <VisibilityOutlined />
                  </Icon>
                </Image.Action>,
              ]}
            />
          </div>
        ))}
      </div>
      <Image.Preview
        list={imgs.map((src) => ({ src }))}
        active={active}
        visible={visible}
        onClose={() => {
          setVisible(false);
        }}
        onActiveChange={setActive}
      />
    </>
  );
}
```
