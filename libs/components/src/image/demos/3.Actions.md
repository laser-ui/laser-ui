# en-US

Use the `actions` prop to set action buttons.

# zh-CN

使用 `actions` 属性设置操作按钮。

```tsx
import { Icon, Image } from '@laser-ui/components';
import { ReactComponent as DeleteOutlined } from '@material-design-icons/svg/outlined/delete.svg';
import { ReactComponent as EditOutlined } from '@material-design-icons/svg/outlined/edit.svg';
import { ReactComponent as MoreHorizOutlined } from '@material-design-icons/svg/outlined/more_horiz.svg';

export default function Demo() {
  return (
    <Image
      style={{ height: 100 }}
      src="/imgs/image-1.jpg"
      actions={[
        <Image.Action title="edit" disabled>
          <Icon>
            <EditOutlined />
          </Icon>
        </Image.Action>,
        <Image.Action title="delete">
          <Icon>
            <DeleteOutlined />
          </Icon>
        </Image.Action>,
        <Image.Action title="more">
          <Icon>
            <MoreHorizOutlined />
          </Icon>
        </Image.Action>,
      ]}
    />
  );
}
```
