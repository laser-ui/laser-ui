# en-US

The `Tabs` has 3 patterns: default, `wrap` and `slider`.

# zh-CN

`Tabs` 有 3 种形态：默认、`wrap` 和 `slider`。

```tsx
import { Compose, Radio, Separator, Tabs } from '@laser-ui/components';
import { useState } from 'react';

export default function Demo() {
  const [placement, setPlacement] = useState('top');
  const [pattern, setPattern] = useState('slider');

  return (
    <>
      <Radio.Group
        list={['top', 'right', 'bottom', 'left'].map((placement) => ({
          label: placement,
          value: placement,
        }))}
        model={placement}
        pattern="outline"
        onModelChange={(value) => {
          setPlacement(value);
        }}
      >
        {(groupProps, optionProps, options) => (
          <Compose {...groupProps} className="mb-3">
            {options.map((opt) => (
              <Radio key={opt.value} {...optionProps(opt)} />
            ))}
          </Compose>
        )}
      </Radio.Group>
      <br />
      <Radio.Group
        list={['slider', 'wrap'].map((pattern) => ({
          label: pattern,
          value: pattern,
        }))}
        model={pattern}
        pattern="outline"
        onModelChange={(value) => {
          setPattern(value);
        }}
      >
        {(groupProps, optionProps, options) => (
          <Compose {...groupProps}>
            {options.map((opt) => (
              <Radio key={opt.value} {...optionProps(opt)} />
            ))}
          </Compose>
        )}
      </Radio.Group>
      <Separator />
      <Tabs
        list={[1, 22, 333, 4444].map((n) => ({
          id: n,
          title: `Tab${n}`,
          panel: `Tab${n} Content...`,
          disabled: n === 333,
        }))}
        placement={placement}
        pattern={pattern}
      />
    </>
  );
}
```
