# en-US

Use the `placement` prop to set the layout.

# zh-CN

使用 `placement` 属性设置布局。

```tsx
import { Compose, Radio, Separator, Tabs } from '@laser-ui/components';
import { useState } from 'react';

export default function Demo() {
  const [placement, setPlacement] = useState('top');

  return (
    <>
      <Radio.Group
        list={['top', 'right', 'bottom', 'left'].map((placement) => ({
          label: placement,
          value: placement,
        }))}
        model={placement}
        pattern="outline"
        onModelChange={(value) => {
          setPlacement(value);
        }}
      >
        {(groupProps, optionProps, options) => (
          <Compose {...groupProps}>
            {options.map((opt) => (
              <Radio key={opt.value} {...optionProps(opt)} />
            ))}
          </Compose>
        )}
      </Radio.Group>
      <Separator />
      <Tabs
        list={[1, 22, 333, 4444].map((n) => ({
          id: n,
          title: `Tab${n}`,
          panel: `Tab${n} Content...`,
          disabled: n === 333,
        }))}
        placement={placement}
      />
    </>
  );
}
```
