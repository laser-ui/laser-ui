# zh-CN

最简单的用法。

```tsx
import { Button, DialogService, Toast } from '@laser-ui/components';

export default function Demo() {
  return (
    <Button
      onClick={() => {
        const { close } = DialogService.open(Toast, {
          children: `Toast message`,
          onClose: () => {
            close();
          },
        });
      }}
    >
      Show Toast
    </Button>
  );
}
```
