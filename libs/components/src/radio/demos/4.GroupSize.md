# en-US

The `Radio.Group` has 3 sizes: `small`, `medium` and `large`.

# zh-CN

`Radio.Group` 有 3 种尺寸：`small`、`medium` 和 `large`。

```tsx
import { Compose, Radio, Separator } from '@laser-ui/components';

const list = [1, 2, 3].map((n) => ({
  label: `Radio ${n}`,
  value: n,
  disabled: n === 2,
}));
export default function Demo() {
  return (
    <>
      <Radio.Group list={list} pattern="outline" size="small">
        {(groupProps, optionProps, options) => (
          <Compose {...groupProps}>
            {options.map((opt) => (
              <Radio key={opt.value} {...optionProps(opt)} />
            ))}
          </Compose>
        )}
      </Radio.Group>
      <Separator />
      <Radio.Group list={list} pattern="outline">
        {(groupProps, optionProps, options) => (
          <Compose {...groupProps}>
            {options.map((opt) => (
              <Radio key={opt.value} {...optionProps(opt)} />
            ))}
          </Compose>
        )}
      </Radio.Group>
      <Separator />
      <Radio.Group list={list} pattern="outline" size="large">
        {(groupProps, optionProps, options) => (
          <Compose {...groupProps}>
            {options.map((opt) => (
              <Radio key={opt.value} {...optionProps(opt)} />
            ))}
          </Compose>
        )}
      </Radio.Group>
    </>
  );
}
```
