# en-US

This shows how to customize dots of the component.

# zh-CN

这里展示了如何自定义组件轴点。

```tsx
import { Icon, Timeline } from '@laser-ui/components';
import { ReactComponent as CancelFilled } from '@material-design-icons/svg/filled/cancel.svg';
import { ReactComponent as CheckCircleOutlined } from '@material-design-icons/svg/outlined/check_circle.svg';

export default function Demo() {
  return (
    <Timeline
      list={[
        {
          content: ['2022-01-01', 'This is a long long long long long text'],
          state: 'completed',
          dot: (
            <Icon>
              <CheckCircleOutlined />
            </Icon>
          ),
        },
        {
          content: ['This is a text', '2022-01-02'],
          state: 'error',
          dot: (
            <Icon size={20}>
              <CancelFilled />
            </Icon>
          ),
        },
        {
          content: ['2022-01-03', 'This is a text'],
          state: 'active',
        },
        {
          content: ['This is a text', '2022-01-04'],
          color: 'rgb(123 74 203)',
        },
      ]}
    />
  );
}
```
