# en-US

Use the `list` prop to add popup buttons.

# zh-CN

使用 `list` 属性添加弹出按钮。

```tsx
import { Fab, Icon } from '@laser-ui/components';
import { ReactComponent as ArrowDropDownOutlined } from '@material-design-icons/svg/outlined/arrow_drop_down.svg';
import { ReactComponent as ArrowDropUpOutlined } from '@material-design-icons/svg/outlined/arrow_drop_up.svg';
import { ReactComponent as ArrowLeftOutlined } from '@material-design-icons/svg/outlined/arrow_left.svg';
import { ReactComponent as ArrowRightOutlined } from '@material-design-icons/svg/outlined/arrow_right.svg';
import { ReactComponent as DeleteOutlined } from '@material-design-icons/svg/outlined/delete.svg';
import { ReactComponent as EditOutlined } from '@material-design-icons/svg/outlined/edit.svg';
import { ReactComponent as FavoriteOutlined } from '@material-design-icons/svg/outlined/favorite.svg';
import { ReactComponent as FileDownloadOutlined } from '@material-design-icons/svg/outlined/file_download.svg';
import { ReactComponent as ShareOutlined } from '@material-design-icons/svg/outlined/share.svg';

const actions = [
  <Fab.Button theme="danger" shape="circle">
    <Icon>
      <DeleteOutlined />
    </Icon>
  </Fab.Button>,
  <Fab.Button pattern="text" shape="circle">
    <Icon>
      <FileDownloadOutlined />
    </Icon>
  </Fab.Button>,
  <Fab.Button pattern="text" shape="circle">
    <Icon>
      <EditOutlined />
    </Icon>
  </Fab.Button>,
];
export default function Demo() {
  return (
    <div style={{ position: 'relative', height: 600 }}>
      {['top', 'right', 'bottom', 'left', 'center'].map((placement) => (
        <Fab
          key={placement}
          style={{
            position: 'absolute',
            ...(placement === 'center'
              ? { top: 'calc(50% - 28px)', left: 'calc(50% - 28px)' }
              : {
                  top: placement === 'right' || placement === 'bottom' ? 20 : undefined,
                  right: placement === 'bottom' || placement === 'left' ? 20 : undefined,
                  bottom: placement === 'top' || placement === 'left' ? 20 : undefined,
                }),
          }}
          list={
            placement === 'center'
              ? [
                  {
                    placement: 'top',
                    actions: [
                      <Fab.Button pattern="text" shape="circle">
                        <Icon>
                          <FavoriteOutlined />
                        </Icon>
                      </Fab.Button>,
                    ],
                  },
                  {
                    placement: 'right',
                    actions: [
                      <Fab.Button pattern="text" shape="circle">
                        <Icon>
                          <FavoriteOutlined />
                        </Icon>
                      </Fab.Button>,
                    ],
                  },
                  {
                    placement: 'bottom',
                    actions: [
                      <Fab.Button pattern="text" shape="circle">
                        <Icon>
                          <FavoriteOutlined />
                        </Icon>
                      </Fab.Button>,
                    ],
                  },
                  {
                    placement: 'left',
                    actions: [
                      <Fab.Button pattern="text" shape="circle">
                        <Icon>
                          <FavoriteOutlined />
                        </Icon>
                      </Fab.Button>,
                    ],
                  },
                ]
              : [
                  {
                    placement,
                    actions,
                  },
                ]
          }
        >
          <Fab.Button shape="circle">
            <Icon>
              {placement === 'top' ? (
                <ArrowDropUpOutlined />
              ) : placement === 'right' ? (
                <ArrowRightOutlined />
              ) : placement === 'bottom' ? (
                <ArrowDropDownOutlined />
              ) : placement === 'left' ? (
                <ArrowLeftOutlined />
              ) : (
                <ShareOutlined />
              )}
            </Icon>
          </Fab.Button>
        </Fab>
      ))}
    </div>
  );
}
```
