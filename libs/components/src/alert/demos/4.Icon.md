# en-US

Use the `icon` prop to set the icon.

# zh-CN

使用 `icon` 属性设置图标。

```tsx
import { Alert, Icon } from '@laser-ui/components';
import { ReactComponent as AccessAlarmOutlined } from '@material-design-icons/svg/outlined/access_alarm.svg';

export default function Demo() {
  const icon = (
    <Icon>
      <AccessAlarmOutlined />
    </Icon>
  );

  return (
    <>
      <Alert icon={icon}>Laser stands for speed and power!</Alert>
      <Alert type="info" icon={icon}>
        Laser stands for speed and power!
      </Alert>
      <Alert title="Info" icon={icon}>
        Laser stands for speed and power!
      </Alert>
      <Alert title="Info" type="info" icon={icon}>
        Laser stands for speed and power!
      </Alert>
    </>
  );
}
```

.l-alert + .l-alert {
  margin-top: 0.5rem;
}
