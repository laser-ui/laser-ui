# en-US

Nested drawers.

# zh-CN

嵌套抽屉。

```tsx
import { Button, Drawer } from '@laser-ui/components';
import { useState } from 'react';

export default function Demo() {
  const [visible1, setVisible1] = useState(false);
  const [visible2, setVisible2] = useState(false);

  return (
    <>
      <Button
        onClick={() => {
          setVisible1(true);
        }}
      >
        Show Drawer1
      </Button>
      <Drawer
        visible={visible1}
        header="Drawer1"
        onClose={() => {
          setVisible1(false);
        }}
      >
        <Button
          onClick={() => {
            setVisible2(true);
          }}
        >
          Show Drawer2
        </Button>
        <Drawer
          visible={visible2}
          header="Drawer2"
          onClose={() => {
            setVisible2(false);
          }}
        >
          <div>Some contents...</div>
          <div>Some contents...</div>
          <div>Some contents...</div>
        </Drawer>
      </Drawer>
    </>
  );
}
```
