# en-US

When the `percent` prop changes, the component shows a transition animation.

# zh-CN

当 `percent` 属性变化时，组件展示过渡动画。

```tsx
import { Button, Compose, Icon, Progress, Separator } from '@laser-ui/components';
import { ReactComponent as AddOutlined } from '@material-design-icons/svg/outlined/add.svg';
import { ReactComponent as RemoveOutlined } from '@material-design-icons/svg/outlined/remove.svg';
import { useState } from 'react';

export default function Demo() {
  const [percent, setPercent] = useState(0);

  return (
    <>
      <Compose>
        <Button
          pattern="secondary"
          icon={
            <Icon>
              <RemoveOutlined />
            </Icon>
          }
          onClick={() => {
            setPercent(Math.max(percent - 10, 0));
          }}
        />
        <Button
          pattern="secondary"
          icon={
            <Icon>
              <AddOutlined />
            </Icon>
          }
          onClick={() => {
            setPercent(Math.min(percent + 10, 100));
          }}
        />
      </Compose>
      <Separator />
      <Progress percent={percent} wave />
      <Progress percent={percent} pattern="circle" />
      <Progress percent={percent} pattern="dashboard" />
    </>
  );
}
```

```scss
.l-progress {
  margin: 0 0.5rem 0.5rem 0;
}
```
